package daily_Problems; import java.util.Arrays; public class Day3SortedSquaredArray { public static void main(String[] args) { // defining an array of integer type // sample one 90, 23, 5, 109, 12, 22, 67, 34 // sample two 1, 2, 3, 5, 6, 8, 9 int[] array = new int[] { 1, 2, 3, 5, 6, 8, 9 }; // invoking sort() method of the Arrays class for (int i = 0; i < array.length; i++) { array[i] = array[i] * array[i]; } Arrays.sort(array); System.out.println(Arrays.toString(array)); } }
